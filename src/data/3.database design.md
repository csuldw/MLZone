### 数据库表设计

当前项目涉及的几张表如下。

## 用户表

```
drop table if exists tb_user;
create table tb_user
(
	id int primary key auto_increment,
	username varchar(32),
	nickname varchar(32),
	password varchar(32),
	sex varchar(8),
	idCard varchar(20),
	birthday varchar(16),
	phone_number varchar(32),
	address varchar(32),
	email varchar(32),
	status varchar(32),
	image varchar(32),
	user_type int,
	account_source int, # 0: 自建; 1:微信; 2:新浪;
	reg_date varchar(32)
)
```

## 文章信息表

```
drop table if exists tb_article;

create table tb_article
(
	id int primary key auto_increment,
	title varchar(64),
	file_path varchar(64),
	user_id int,
	keywords varchar(255),
	tags varchar(255),
	article_type int, 	#0 原创；1 转载 
	public_date varchar(32)
)
```

## 评论表

```
drop table if exists tb_comment;

create table tb_comment
(
	id int primary key auto_increment,
	content varchar(255),
	from_user_id int,
	to_user_id int,
	article_id int,
	parent_comment_id int, 
	send_date varchar(32)
)
```

## 点赞表

```
drop table if exists tb_star;

create table tb_star
(
	id int primary key auto_increment,
	from_user_id int,
	to_user_id int,
	comment_id int,
	star_date varchar(32)
)
```

## 推荐表

```
drop table if exists tb_recommendation;
create table tb_recommendation
(
	id int primary key auto_increment,
	title varchar(32), 			# 标题
	link varchar(32),  			# 链接
	source varchar(32), 		#来源
	author varchar(16),			#作者
	recommendation_type int, 	#推荐类型（0: 优秀文章链接推荐; 1: 牛人博客推荐; 2：其他）
	recommend_date varchar(32)
)
```
